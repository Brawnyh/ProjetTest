#@ function cubes_carre(n:int)->int = (n*n *( n+1)*(n+1))// 4


def cubes(n:int) -> int:
    #@ requires n>=0
    #@ ensures result == cubes_carre(n)
    t=0
    for i in range(n):
        t+=i*i*i
    return t
 
r = cubes(0)
#@ assert (r==1)
r= cubes ( 1 )
