#include <stdio.h>

void AfficherTab(int Tab[], int NbElt) {
    int i;
    for (i = 0; i < NbElt; i++) {
        printf("%d ", Tab[i]);
    }
    printf("\n");
}

int Min(int Tab[], int NbElt) {
    int i, min = Tab[0];
    for (i = 1; i < NbElt; i++) {
        if (Tab[i] < min) {
            min = Tab[i];
        }
    }
    return min;
}

int Max(int Tab[], int NbElt) {
    int i, max = Tab[0];
    for (i = 1; i < NbElt; i++) {
        if (Tab[i] > max) {
            max = Tab[i];
        }
    }
    return max;
}

void MinMax(int Tab[], int NbElt, int *pMin, int *pMax) {
    int i;
    *pMin = Tab[0];
    *pMax = Tab[0];
    for (i = 1; i < NbElt; i++) {
        if (Tab[i] < *pMin) {
            *pMin = Tab[i];
        }
        if (Tab[i] > *pMax) {
            *pMax = Tab[i];
        }
    }
}

int SupprimerVal(int Tab[], int *pNbElt, int Val) {
    int i, j = 0, suppr = 0;
    for (i = 0; i < *pNbElt; i++) {
        if (Tab[i] != Val) {
            Tab[j] = Tab[i];
            j++;
        } else {
            suppr++;
        }
    }
    *pNbElt = j;
    return suppr;
}

int main() {
    int Tab[100];
    int n, i, val, nbSuppr;
    int min, max;

    printf("Combien d'elements ? (max 100) : ");
    scanf("%d", &n);

    for (i = 0; i < n; i++) {
        printf("Tab[%d] = ", i);
        scanf("%d", &Tab[i]);
    }

    printf("Tableau : ");
    AfficherTab(Tab, n);

    min = Min(Tab, n);
    max = Max(Tab, n);
    printf("Min = %d, Max = %d\n", min, max);

    MinMax(Tab, n, &min, &max);
    printf("(MinMax) Min = %d, Max = %d\n", min, max);

    printf("Valeur a supprimer : ");
    scanf("%d", &val);

    nbSuppr = SupprimerVal(Tab, &n, val);
    printf("%d occurrences supprimees.\n", nbSuppr);

    printf("Tableau apres suppression : ");
    AfficherTab(Tab, n);

    return 0;
}Hello world



#include <stdio.h>

void Convertir1(int secondes, int *heures, int *minutes, int *secs) {
    *heures = secondes / 3600;
    secondes = secondes % 3600;
    *minutes = secondes / 60;
    *secs = secondes % 60;
}

void Convertir2(int secondes, int tab[3]) {
    tab[0] = secondes / 3600;
    secondes = secondes % 3600;
    tab[1] = secondes / 60;
    tab[2] = secondes % 60;
}

int main() {
    int secondes;
    int h, m, s;
    int tab[3];

    printf("Entrez une duree en secondes : ");
    scanf("%d", &secondes);

    Convertir1(secondes, &h, &m, &s);
    printf("Avec Convertir1 : %d h %d min %d s\n", h, m, s);

    Convertir2(secondes, tab);
    printf("Avec Convertir2 : %d h %d min %d s\n", tab[0], tab[1], tab[2]);

    return 0;
}


#include <stdio.h>

void MinMaxAdresse(int Tab[], int NbElt, int **pAdrMin, int **pAdrMax) {
    int i;
    *pAdrMin = &Tab[0];
    *pAdrMax = &Tab[0];

    for (i = 1; i < NbElt; i++) {
        if (Tab[i] < **pAdrMin) {
            *pAdrMin = &Tab[i];
        }
        if (Tab[i] > **pAdrMax) {
            *pAdrMax = &Tab[i];
        }
    }
}

int main() {
    int Tab[100];
    int n, i;
    int *adrMin, *adrMax;

    printf("Combien d'elements ? (max 100) : ");
    scanf("%d", &n);

    for (i = 0; i < n; i++) {
        printf("Tab[%d] = ", i);
        scanf("%d", &Tab[i]);
    }

    MinMaxAdresse(Tab, n, &adrMin, &adrMax);

    printf("Adresse du min : %p -> valeur = %d\n", (void *)adrMin, *adrMin);
    printf("Adresse du max : %p -> valeur = %d\n", (void *)adrMax, *adrMax);

    return 0;
}
